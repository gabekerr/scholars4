Theorem 1.2.11.

Suppose \(f : U \to \mathbb{R}\) is a scalar function on an open subset \(U\) of \(\mathbb{R}^m\text{.}\) If \(f\) has continuous first partial derivatives then \(f\) is differentiable. Furthermore, in this case
\begin{equation*} \tder{}{f} \left( \threevec{b_1}{\vdots}{b_m} \right) = b_1 \pd{f}{x_1} + b_2 \pd{f}{x_2} + \cdots + b_m \pd{f}{x_m}. \end{equation*}

Proof.

\(U\)\(\mb{a}\)\(\mb{x}\)\(U\text{.}\)\(\mb{x} - \mb{a} = b_1 \mb{e}_1 + \cdots + b_m \mb{e}_m\text{.}\)
\begin{align*} \mb{v}_1 \amp = b_1 \mb{e}_1 + \cdots + b_m \mb{e}_m, \\ \mb{v}_2 \amp = b_2 \mb{e}_2 + \cdots + b_m \mb{e}_m, \\ \amp \vdots \\ \mb{v}_m \amp = b_m \mb{e}_m. \end{align*}
\begin{align*} f(\mb{x} ) - f (\mb{a}) \amp = \left( f(\mb{a} + \mb{v}_1 ) - f (\mb{a} + \mb{v}_2) \right) + \cdots + \left( f(\mb{a} + \mb{v}_m ) - f (\mb{a}) \right), \\ \amp = \left( f(\mb{a} + \mb{v}_2 + b_1 \mb{e}_1 ) - f (\mb{a} + \mb{v}_2) \right) + \cdots + \left( f(\mb{a} + b_m \mb{e}_m ) - f (\mb{a}) \right), \end{align*}
\begin{equation*} f(\mb{a} + \mb{v}_{k + 1} + b_k \mb{e}_k ) - f(\mb{a} + \mb{v}_{k + 1}). \end{equation*}
\(k\)\(t_k\)\(0\)\(b_k\)
\begin{equation*} \left. \pd{f}{x_k} \right|_{\mb{a} + t_k \mb{e}_{k}} = \frac{f(\mb{a} + \mb{v}_{k + 1} + b_k \mb{e}_k ) - f(\mb{a} + \mb{v}_{k + 1})}{b_k} \end{equation*}
\begin{equation*} f(\mb{a} + \mb{v}_{k + 1} + b_k \mb{e}_k ) - f(\mb{a} + \mb{v}_{k + 1}) = b_k \left. \pd{f}{x_k} \right|_{\mb{a} + \mb{v}_{k + 1} + t_k \mb{e}_{k}}. \end{equation*}
\begin{align*} f(\mb{x} ) - f (\mb{a}) \amp = \left( f(\mb{a} + \mb{v}_2 + b_1 \mb{e}_1 ) - f (\mb{a} + \mb{v}_2) \right) + \cdots + \left( f(\mb{a} + b_m \mb{e}_m ) - f (\mb{a}) \right), \\ \amp = b_1 \left. \pd{f}{x_1} \right|_{\mb{a} + \mb{v}_{2} + t_1 \mb{e}_{1}} + \cdots + b_m \left. \pd{f}{x_m} \right|_{\mb{a} + t_m \mb{e}_{m}} \end{align*}
\begin{equation*} L (\mb{x} - \mb{a} ) = b_1 \left. \pd{f}{x_1} \right|_{\mb{a}} + \cdots + b_m \left. \pd{f}{x_m} \right|_{\mb{a}} \end{equation*}
\(f\)\(\mb{a}\text{.}\)
\begin{equation*} N (\mb{x} ) = f(\mb{x} ) - f (\mb{a}) - L (\mb{x} - \mb{a} ) \end{equation*}
(1.2.1)
\begin{align*} N(\mb{x}) \amp = b_1 \left( \left. \pd{f}{x_1} \right|_{\mb{a} + \mb{v}_{2} + t_1 \mb{e}_{1}} - \left. \pd{f}{x_1} \right|_{\mb{a}} \right) + \cdots + b_m \left( \left. \pd{f}{x_m} \right|_{\mb{a} + t_m \mb{e}_{m}} - \left. \pd{f}{x_m} \right|_{\mb{a}} \right), \\ \amp = \left( \mb{x} - \mb{a} \right) \cdot \threevec{\left. \pd{f}{x_1} \right|_{\mb{a} + \mb{v}_{2} + t_1 \mb{e}_{1}} - \left. \pd{f}{x_1} \right|_{\mb{a}} }{\vdots}{ \left. \pd{f}{x_m} \right|_{\mb{a} + t_m \mb{e}_{m}} - \left. \pd{f}{x_m} \right|_{\mb{a}} }. \end{align*}
\(\mb{x}\)\(\mb{a}\text{,}\)\(t_k\)\(b_k\)\(\mb{v}_k\text{.}\)
\begin{equation*} \lim_{\mb{x} \to \mb{a}} \threevec{\left. \pd{f}{x_1} \right|_{\mb{a} + \mb{v}_{2} + t_1 \mb{e}_{1}} - \left. \pd{f}{x_1} \right|_{\mb{a}} }{\vdots}{ \left. \pd{f}{x_m} \right|_{\mb{a} + t_m \mb{e}_{m}} - \left. \pd{f}{x_m} \right|_{\mb{a}} } = \mb{0}. \end{equation*}
\begin{equation*} \lim_{\mb{x} \to \mb{a}} \left\| \threevec{\left. \pd{f}{x_1} \right|_{\mb{a} + \mb{v}_{2} + t_1 \mb{e}_{1}} - \left. \pd{f}{x_1} \right|_{\mb{a}} }{\vdots}{ \left. \pd{f}{x_m} \right|_{\mb{a} + t_m \mb{e}_{m}} - \left. \pd{f}{x_m} \right|_{\mb{a}} } \right\| = 0 \end{equation*}
\(\mb{x} - \mb{a}\)\(\| \mb{x} - \mb{a} \|\)\(\mb{u}\)
\begin{equation*} \mb{u} = \frac{1}{\| \mb{v}_1\|} \mb{v}_1 = \frac{\mb{x} - \mb{a}}{\| \mb{x} - \mb{a}\|} \end{equation*}
\begin{align*} \frac{N(\mb{x})}{\| \mb{x} - \mb{a}\|} \amp = \left( \frac{\mb{x} - \mb{a}}{\|\mb{x} - \mb{a} \|} \right) \cdot \threevec{\left. \pd{f}{x_1} \right|_{\mb{a} + \mb{v}_{2} + t_1 \mb{e}_{1}} - \left. \pd{f}{x_1} \right|_{\mb{a}} }{\vdots}{ \left. \pd{f}{x_m} \right|_{\mb{a} + t_m \mb{e}_{m}} - \left. \pd{f}{x_m} \right|_{\mb{a}} }, \\ \amp = \cos \theta_{\mb{x}} \|\mb{u} \| \left\| \threevec{\left. \pd{f}{x_1} \right|_{\mb{a} + \mb{v}_{2} + t_1 \mb{e}_{1}} - \left. \pd{f}{x_1} \right|_{\mb{a}} }{\vdots}{ \left. \pd{f}{x_m} \right|_{\mb{a} + t_m \mb{e}_{m}} - \left. \pd{f}{x_m} \right|_{\mb{a}} } \right\|, \\ \amp = \cos \theta_{\mb{x}} \left\| \threevec{\left. \pd{f}{x_1} \right|_{\mb{a} + \mb{v}_{2} + t_1 \mb{e}_{1}} - \left. \pd{f}{x_1} \right|_{\mb{a}} }{\vdots}{ \left. \pd{f}{x_m} \right|_{\mb{a} + t_m \mb{e}_{m}} - \left. \pd{f}{x_m} \right|_{\mb{a}} } \right\|. \end{align*}
\begin{equation*} \lim_{\mb{x} \to \mb{a}} \frac{N(\mb{x})}{\| \mb{x} - \mb{a}\|} = 0 \end{equation*}
\(\tder{\mb{a}}{f} = L\)
